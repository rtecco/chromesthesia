export function initDebugPanel(onMasterGainChange: (v: number) => void): void {
  const toggle = document.createElement('button');
  toggle.type = 'button';
  toggle.className = 'debug-toggle';
  toggle.textContent = '\u2699'; // gear icon
  toggle.title = 'Audio debug parameters';

  const panel = document.createElement('div');
  panel.className = 'debug-panel hidden';

  const heading = document.createElement('div');
  heading.className = 'debug-heading';
  heading.textContent = 'Audio Parameters';
  panel.appendChild(heading);

  for (const def of SLIDERS) {
    const row = document.createElement('div');
    row.className = 'debug-row';

    const label = document.createElement('label');
    label.className = 'debug-label';
    label.textContent = def.label;

    const value = document.createElement('span');
    value.className = 'debug-value';
    value.textContent = String(debugParams[def.key]);

    const input = document.createElement('input');
    input.type = 'range';
    input.className = 'debug-slider';
    input.min = String(def.min);
    input.max = String(def.max);
    input.step = String(def.step);
    input.value = String(debugParams[def.key]);

    input.addEventListener('input', () => {
      const v = parseFloat(input.value);
      (debugParams as Record<string, number>)[def.key] = v;
      value.textContent = def.step >= 1 ? String(v) : v.toFixed(2);
      if (def.key === 'masterGain') onMasterGainChange(v);
    });

    row.appendChild(label);
    row.appendChild(input);
    row.appendChild(value);
    panel.appendChild(row);
  }

  // Export button — logs current values to console for easy copy
  const exportBtn = document.createElement('button');
  exportBtn.type = 'button';
  exportBtn.className = 'debug-export';
  exportBtn.textContent = 'Log values to console';
  exportBtn.addEventListener('click', () => {
    console.log('Debug params:', JSON.stringify(debugParams, null, 2));
  });
  panel.appendChild(exportBtn);

  toggle.addEventListener('click', () => {
    panel.classList.toggle('hidden');
  });

  document.getElementById('app')!.appendChild(toggle);
  document.getElementById('app')!.appendChild(panel);
}

Explain the color / sound model a bit more to me.

Assistant2026-03-23T17:39:42.763Z

The mapping works in three layers:

1. Color → Base Voice (HSL)

The color's HSL values determine the fundamental character of the sound:

  • Hue (what color it is) → waveform character. The hue is converted to a "warmth" score. Reds/oranges/yellows are warm → sawtooth waves (buzzy, brassy). Blues/purples are cool → triangle/sine waves (hollow, glassy). This also affects the filter type: warm hues get low-pass (muffled richness), cool hues get band-pass or high-pass (airy, open).

  • Saturation (how vivid) → harmonic richness. High saturation = more oscillator partials (up to 4), wider detuning between them (chorus-like fatness), more noise "breath". A pale lavender sounds pure and thin. A vivid cobalt blue sounds thicker and more complex.

  • Lightness (how bright/dark) → pitch. Dark colors are low-pitched (down to 55Hz), light colors are high-pitched (up to 880Hz). Titanium White is almost at the top of the range; a deep navy would rumble.

2. Position → Modulation (X/Y on canvas)

Where you paint shifts the voice in real-time:

  • X-axis → filter cutoff (brightness). Far left is very muffled/dark, far right is wide open/bright. This is an exponential curve so the middle of the canvas is fairly neutral.

  • Y-axis → pitch shift + reverb. Top of canvas shifts pitch up (up to +14 semitones) and is dry. Bottom shifts pitch down and drenches in reverb. So the top-right corner is the brightest, highest, driest point. Bottom-left is the darkest, lowest, most reverberant.

3. Brush → Effects Chain

The brush type adds a fixed effect on top of the voice:

  • Flat → reverb (cathedral wash)
  • Round → chorus (shimmery doubling)
  • Knife → distortion + slap delay (aggressive, edgy)
  • Dry → fast tremolo (gritty stutter)

In practice

So if you pick Cadmium Red (warm hue, high saturation, medium lightness) and paint with the Flat brush in the top-right corner: you get a mid-pitched buzzy sawtooth with lots of detuned partials, filter wide open, pitched up, barely any reverb, through a cathedral reverb effect.

Same color, bottom-left corner: same base voice but now muffled, pitched down over an octave, swimming in reverb.

Switch to Lavender (cool hue, low saturation, high lightness): now it's a high-pitched pure sine/triangle, thin, glassy — completely different character from the same position.
