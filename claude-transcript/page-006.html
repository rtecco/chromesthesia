    strokeIndex: number;
    startOffset: number;   // seconds from loop start
    duration: number;      // seconds
    points: readonly StrokePoint[];
  };

  let schedule: ScheduledStroke[] = [];
  let nextEvent: { strokeIdx: number; pointIdx: number } = { strokeIdx: 0, pointIdx: 0 };

  // Track active voices for cleanup and voice stealing
  type ActiveReplayVoice = {
    voice: Voice;
    stopTime: number;
  };
  const activeVoices: ActiveReplayVoice[] = [];

  function buildSchedule(): ScheduledStroke[] {
    const painting = getPainting();
    const loopDur = painting.loopLengthMs / 1000;
    const strokes = painting.strokes.filter((s) => !ERASERS.has(s.brush) && s.points.length >= 2);

    if (strokes.length === 0) return [];

    // Measure each stroke's recorded duration
    const rawDurations = strokes.map((s) => {
      const pts = s.points;
      return Math.max(pts[pts.length - 1].timestamp - pts[0].timestamp, 50) / 1000;
    });
    const totalRawDur = rawDurations.reduce((a, b) => a + b, 0);

    // Gap between strokes: 5% of loop time, split evenly
    const totalGap = loopDur * 0.05;
    const gapPerStroke = strokes.length > 1 ? totalGap / (strokes.length - 1) : 0;
    const availableTime = loopDur - totalGap;

    // Scale each stroke's duration proportionally to fill available time
    const result: ScheduledStroke[] = [];
    let offset = 0;
    for (let i = 0; i < strokes.length; i++) {
      const scaledDur = (rawDurations[i] / totalRawDur) * availableTime;
      result.push({
        strokeIndex: i,
        startOffset: offset,
        duration: scaledDur,
        points: strokes[i].points,
      });
      offset += scaledDur + gapPerStroke;
    }

    return result;
  }

  function getLoopDuration(): number {
    return getPainting().loopLengthMs / 1000;
  }

  /** Returns position within current loop in seconds */
  function loopPosition(): number {
    const ctx = getAudioContext();
    const elapsed = ctx.currentTime - loopStartTime;
    const loopDur = getLoopDuration();
    return elapsed % loopDur;
  }

  function triggerStrokeVoice(
    ss: ScheduledStroke,
    pointIdx: number,
    when: number,
  ) {
    const painting = getPainting();
    const strokes = painting.strokes.filter((s) => !ERASERS.has(s.brush) && s.points.length >= 2);
    const stroke = strokes[ss.strokeIndex];
    if (!stroke) return;

    const point = ss.points[pointIdx];
    const params = voiceParamsFromColor(stroke.color);
    const mod = positionMod(point);

    // Voice stealing
    while (activeVoices.length >= debugParams.maxVoices) {
      const oldest = activeVoices.shift()!;
      oldest.voice.stop();
    }

    const voice = createVoice(params, mod, point.pressure);
    const chain = getEffectChain(stroke.brush);
    voice.output.connect(chain.input);

    // Voice duration: time until next point, scaled to scheduled duration
    const rawDur = (ss.points[ss.points.length - 1].timestamp - ss.points[0].timestamp) / 1000;
    const timeScale = rawDur > 0 ? ss.duration / rawDur : 1;
    let voiceDur: number;
    if (pointIdx < ss.points.length - 1) {
      voiceDur = ((ss.points[pointIdx + 1].timestamp - point.timestamp) / 1000) * timeScale;
      voiceDur = Math.max(voiceDur, 0.02);
    } else {
      voiceDur = 0.08; // last point — short tail
    }

    const stopTime = when + voiceDur * debugParams.voiceOverlap;
    voice.stop(stopTime);
    activeVoices.push({ voice, stopTime });
  }

  function scheduleAhead() {
    if (schedule.length === 0) return;

    const ctx = getAudioContext();
    const now = ctx.currentTime;
    const loopDur = getLoopDuration();
    const lookahead = LOOKAHEAD_MS / 1000;

    // Figure out where we are in the loop
    const elapsed = now - loopStartTime;
    const loopIter = Math.floor(elapsed / loopDur);
    const loopBase = loopStartTime + loopIter * loopDur;

    // Walk through schedule and trigger anything within the lookahead window
    for (let si = nextEvent.strokeIdx; si < schedule.length; si++) {
      const ss = schedule[si];
      const rawDur = (ss.points[ss.points.length - 1].timestamp - ss.points[0].timestamp) / 1000;
      const timeScale = rawDur > 0 ? ss.duration / rawDur : 1;

      const startPt = si === nextEvent.strokeIdx ? nextEvent.pointIdx : 0;
      for (let pi = startPt; pi < ss.points.length; pi++) {
        // When should this point play?
        const pointOffsetInStroke = ((ss.points[pi].timestamp - ss.points[0].timestamp) / 1000) * timeScale;
        let absTime = loopBase + ss.startOffset + pointOffsetInStroke;

        // If it's already past in this loop iteration, try next
        if (absTime < now - 0.01) {
          absTime += loopDur;
        }

        if (absTime > now + lookahead) {
          // Save position for next scheduling pass
          nextEvent = { strokeIdx: si, pointIdx: pi };
          return;
        }

        if (absTime >= now - 0.01) {
          triggerStrokeVoice(ss, pi, Math.max(absTime, now));
        }
      }
    }

    // We've reached the end of the schedule — wrap to next loop
    nextEvent = { strokeIdx: 0, pointIdx: 0 };
    loopStartTime = loopBase + loopDur;

    // Cleanup finished voices
    const cutoff = now - 0.5;
    for (let i = activeVoices.length - 1; i >= 0; i--) {
      if (activeVoices[i].stopTime < cutoff) {
        activeVoices.splice(i, 1);
      }
    }
  }

  function updateVisual() {
    if (!playing) return;
    const loopDur = getLoopDuration();
    const pos = loopPosition();

    // Find which stroke is currently playing and use its average X for the playhead
    let visualX = pos / loopDur; // fallback: linear sweep

    for (const ss of schedule) {
      if (pos >= ss.startOffset && pos <= ss.startOffset + ss.duration) {
        // We're inside this stroke — interpolate X from its points
        const t = (pos - ss.startOffset) / ss.duration;
        const pi = Math.min(Math.floor(t * (ss.points.length - 1)), ss.points.length - 1);
        visualX = ss.points[pi].x;
        break;
      }
    }

    onPositionChange(visualX);
    animFrameId = requestAnimationFrame(updateVisual);
  }

  function stopAllVoices() {
    for (const v of activeVoices) {
      v.voice.stop();
    }
    activeVoices.length = 0;
  }

  return {
    start() {
      if (playing) return;
      playing = true;
      void ensureResumed();

      schedule = buildSchedule();
      nextEvent = { strokeIdx: 0, pointIdx: 0 };

      const ctx = getAudioContext();
      loopStartTime = ctx.currentTime;

      schedulerTimer = setInterval(scheduleAhead, SCHEDULE_INTERVAL_MS);
      scheduleAhead();
      animFrameId = requestAnimationFrame(updateVisual);
    },

    stop() {
      if (!playing) return;
      playing = false;
      if (schedulerTimer !== null) {
        clearInterval(schedulerTimer);
        schedulerTimer = null;
      }
      if (animFrameId) {
        cancelAnimationFrame(animFrameId);
        animFrameId = 0;
      }
      stopAllVoices();
      schedule = [];
      onPositionChange(-1);
    },

    isPlaying: () => playing,
    getPosition() {
      if (!playing) return 0;
      return loopPosition() / getLoopDuration();
    },
  };
