import type { BrushType, EraserTool, Painting, StrokePoint } from '../types';
import { getAudioContext, ensureResumed } from './engine';
import { createVoice, type Voice } from './voices';
import { voiceParamsFromColor, positionMod } from './mappings';
import { getEffectChain } from './effects';
import { debugParams } from '../ui/debug';

const ERASERS: ReadonlySet<BrushType> = new Set<EraserTool>(['scraper', 'solvent']);
const LOOKAHEAD_MS = 80;
const SCHEDULE_INTERVAL_MS = 30;

export type Playhead = {
  start: () => void;
  stop: () => void;
  isPlaying: () => boolean;
};

/**
 * Stroke-replay: replays each stroke's recorded trajectory as sound,
 * in the order they were painted, at their original speed, looping.
 */
export function createPlayhead(
  getPainting: () => Painting,
): Playhead {
  let playing = false;
  let schedulerTimer: ReturnType<typeof setInterval> | null = null;
  let loopStartTime = 0;
  let loopDuration = 0; // computed from strokes

  type ScheduledStroke = {
    strokeIndex: number;
    startOffset: number;   // seconds from loop start
    duration: number;      // seconds
    points: readonly StrokePoint[];
  };

  let schedule: ScheduledStroke[] = [];
  let nextEvent = { strokeIdx: 0, pointIdx: 0 };

  type ActiveReplayVoice = {
    voice: Voice;
    stopTime: number;
  };
  const activeVoices: ActiveReplayVoice[] = [];

  function buildSchedule(): { items: ScheduledStroke[]; totalDuration: number } {
    const painting = getPainting();
    const strokes = painting.strokes.filter((s) => !ERASERS.has(s.brush) && s.points.length >= 2);

    if (strokes.length === 0) return { items: [], totalDuration: 0 };

    const GAP = 0.08; // seconds between strokes
    const items: ScheduledStroke[] = [];
    let offset = 0;

    for (let i = 0; i < strokes.length; i++) {
      const pts = strokes[i].points;
      const dur = Math.max(pts[pts.length - 1].timestamp - pts[0].timestamp, 50) / 1000;
      items.push({
        strokeIndex: i,
        startOffset: offset,
        duration: dur,
        points: pts,
      });
      offset += dur + GAP;
    }

    const totalDuration = offset - (strokes.length > 1 ? GAP : 0);
    return { items, totalDuration };
  }

  function triggerStrokeVoice(
    ss: ScheduledStroke,
    pointIdx: number,
    when: number,
  ) {
    const painting = getPainting();
    const strokes = painting.strokes.filter((s) => !ERASERS.has(s.brush) && s.points.length >= 2);
    const stroke = strokes[ss.strokeIndex];
    if (!stroke) return;

    const point = ss.points[pointIdx];
    const params = voiceParamsFromColor(stroke.color);
    const mod = positionMod(point);

    // Voice stealing
    while (activeVoices.length >= debugParams.maxVoices) {
      const oldest = activeVoices.shift()!;
      oldest.voice.stop();
    }

    const voice = createVoice(params, mod, point.pressure);
    const chain = getEffectChain(stroke.brush);
    voice.output.connect(chain.input);

    // Voice duration: time until next point
    let voiceDur: number;
    if (pointIdx < ss.points.length - 1) {
      voiceDur = (ss.points[pointIdx + 1].timestamp - point.timestamp) / 1000;
      voiceDur = Math.max(voiceDur, 0.02);
    } else {
      voiceDur = 0.08; // last point — short tail
    }

    const stopTime = when + voiceDur * debugParams.voiceOverlap;
    voice.stop(stopTime);
    activeVoices.push({ voice, stopTime });
  }

  function scheduleAhead() {
    if (schedule.length === 0) return;

    const ctx = getAudioContext();
    const now = ctx.currentTime;
    const lookahead = LOOKAHEAD_MS / 1000;

    const elapsed = now - loopStartTime;
    const loopIter = Math.floor(elapsed / loopDuration);
    const loopBase = loopStartTime + loopIter * loopDuration;

    for (let si = nextEvent.strokeIdx; si < schedule.length; si++) {
      const ss = schedule[si];

      const startPt = si === nextEvent.strokeIdx ? nextEvent.pointIdx : 0;
      for (let pi = startPt; pi < ss.points.length; pi++) {
        const pointOffset = (ss.points[pi].timestamp - ss.points[0].timestamp) / 1000;
        let absTime = loopBase + ss.startOffset + pointOffset;

        if (absTime < now - 0.01) {
          absTime += loopDuration;
        }

        if (absTime > now + lookahead) {
          nextEvent = { strokeIdx: si, pointIdx: pi };
          return;
        }

        if (absTime >= now - 0.01) {
          triggerStrokeVoice(ss, pi, Math.max(absTime, now));
        }
      }
    }

    // Wrap to next loop
    nextEvent = { strokeIdx: 0, pointIdx: 0 };
    loopStartTime = loopBase + loopDuration;

    // Cleanup finished voices
    const cutoff = now - 0.5;
    for (let i = activeVoices.length - 1; i >= 0; i--) {
      if (activeVoices[i].stopTime < cutoff) {
        activeVoices.splice(i, 1);
      }
    }
  }

  function stopAllVoices() {
    for (const v of activeVoices) {
      v.voice.stop();
    }
    activeVoices.length = 0;
  }

  return {
    start() {
      if (playing) return;
      playing = true;
      void ensureResumed();

      const built = buildSchedule();
      schedule = built.items;
      loopDuration = built.totalDuration;
      if (loopDuration <= 0) {
        playing = false;
        return;
      }

      nextEvent = { strokeIdx: 0, pointIdx: 0 };
      loopStartTime = getAudioContext().currentTime;

      schedulerTimer = setInterval(scheduleAhead, SCHEDULE_INTERVAL_MS);
      scheduleAhead();
    },

    stop() {
      if (!playing) return;
      playing = false;
      if (schedulerTimer !== null) {
        clearInterval(schedulerTimer);
        schedulerTimer = null;
      }
      stopAllVoices();
      schedule = [];
    },

    isPlaying: () => playing,
  };
}

import type { BrushType, EraserTool, Painting, StrokePoint } from '../types';
import { getAudioContext, ensureResumed } from './engine';
import { createVoice, type Voice } from './voices';
import { voiceParamsFromColor, positionMod } from './mappings';
import { getEffectChain } from './effects';
import { debugParams } from '../ui/debug';

const ERASERS: ReadonlySet<BrushType> = new Set<EraserTool>(['scraper', 'solvent']);
const LOOKAHEAD_MS = 80;
const SCHEDULE_INTERVAL_MS = 30;

export type Playhead = {
  start: () => void;
  stop: () => void;
  isPlaying: () => boolean;
};

/**
 * Stroke-replay: replays each stroke as a single continuous voice
 * (matching live audition behavior), updating position along the
 * recorded trajectory. Loops with a pause at the end.
 */
export function createPlayhead(
  getPainting: () => Painting,
): Playhead {
  let playing = false;
  let schedulerTimer: ReturnType<typeof setInterval> | null = null;
  let loopStartTime = 0;
  let loopDuration = 0;

  type ScheduledStroke = {
    strokeIndex: number;
    startOffset: number;   // seconds from loop start
    duration: number;      // seconds
    points: readonly StrokePoint[];
  };

  let schedule: ScheduledStroke[] = [];
  let nextStrokeIdx = 0;

  type ActiveStrokeVoice = {
    voice: Voice;
    ss: ScheduledStroke;
    voiceStartTime: number; // absolute AudioContext time when voice started
    nextPointIdx: number;
    stopTime: number;
  };
  const activeVoices: ActiveStrokeVoice[] = [];

  function buildSchedule(): { items: ScheduledStroke[]; totalDuration: number } {
    const painting = getPainting();
    const strokes = painting.strokes.filter((s) => !ERASERS.has(s.brush) && s.points.length >= 2);

    if (strokes.length === 0) return { items: [], totalDuration: 0 };

    const GAP = 0.08;
    const items: ScheduledStroke[] = [];
    let offset = 0;

    for (let i = 0; i < strokes.length; i++) {
      const pts = strokes[i].points;
      const dur = Math.max(pts[pts.length - 1].timestamp - pts[0].timestamp, 50) / 1000;
      items.push({
        strokeIndex: i,
        startOffset: offset,
        duration: dur,
        points: pts,
      });
      offset += dur + GAP;
    }

    const totalDuration = offset - (strokes.length > 1 ? GAP : 0) + 2.5;
    return { items, totalDuration };
  }

  function startStrokeVoice(ss: ScheduledStroke, when: number) {
    const painting = getPainting();
    const strokes = painting.strokes.filter((s) => !ERASERS.has(s.brush) && s.points.length >= 2);
    const stroke = strokes[ss.strokeIndex];
    if (!stroke) return;

    // Voice stealing
    while (activeVoices.length >= debugParams.maxVoices) {
      const oldest = activeVoices.shift()!;
      oldest.voice.stop();
    }

    const firstPoint = ss.points[0];
    const params = voiceParamsFromColor(stroke.color);
    const mod = positionMod(firstPoint);
    const voice = createVoice(params, mod, 0.5);
    const chain = getEffectChain(stroke.brush);
    voice.output.connect(chain.input);

    const stopTime = when + ss.duration;
    voice.stop(stopTime);

    activeVoices.push({ voice, ss, voiceStartTime: when, nextPointIdx: 1, stopTime });
  }

  function scheduleAhead() {
    if (schedule.length === 0) return;

    const ctx = getAudioContext();
    const now = ctx.currentTime;
    const lookahead = LOOKAHEAD_MS / 1000;

    // Check if we need to start a new loop
    if (now >= loopStartTime + loopDuration - lookahead && nextStrokeIdx >= schedule.length) {
      // Reset for the next loop
      loopStartTime += loopDuration;
      nextStrokeIdx = 0;
    }

    // Start new stroke voices that fall within the lookahead window
    while (nextStrokeIdx < schedule.length) {
      const ss = schedule[nextStrokeIdx];
      const absTime = loopStartTime + ss.startOffset;

      if (absTime > now + lookahead) break;

      if (absTime >= now - 0.05) {
        startStrokeVoice(ss, Math.max(absTime, now));
      }
      nextStrokeIdx++;
    }

    // Update position on active voices using their own start time
    for (const av of activeVoices) {
      const strokeElapsed = now - av.voiceStartTime;
      if (strokeElapsed < 0) continue;

      while (av.nextPointIdx < av.ss.points.length) {
        const pointTime = (av.ss.points[av.nextPointIdx].timestamp - av.ss.points[0].timestamp) / 1000;
        if (pointTime > strokeElapsed) break;

        const mod = positionMod(av.ss.points[av.nextPointIdx]);
        av.voice.updatePosition(mod);
        av.nextPointIdx++;
      }
    }

    // Cleanup finished voices
    for (let i = activeVoices.length - 1; i >= 0; i--) {
      if (activeVoices[i].stopTime < now - 0.5) {
        activeVoices.splice(i, 1);
      }
    }
  }

  function stopAllVoices() {
    for (const v of activeVoices) {
      v.voice.stop();
    }
    activeVoices.length = 0;
  }

  return {
    start() {
      if (playing) return;
      playing = true;
      void ensureResumed();

      const built = buildSchedule();
      schedule = built.items;
      loopDuration = built.totalDuration;
      if (loopDuration <= 0) {
        playing = false;
        return;
      }

      nextStrokeIdx = 0;
      loopStartTime = getAudioContext().currentTime;

      schedulerTimer = setInterval(scheduleAhead, SCHEDULE_INTERVAL_MS);
      scheduleAhead();
    },

    stop() {
      if (!playing) return;
      playing = false;
      if (schedulerTimer !== null) {
        clearInterval(schedulerTimer);
        schedulerTimer = null;
      }
      stopAllVoices();
      schedule = [];
    },

    isPlaying: () => playing,
  };
}
